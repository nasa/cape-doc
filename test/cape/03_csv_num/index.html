#!/usr/bin/env python
# -*- coding: utf-8 -*-

# Import CSV module
import cape.attdb.ftypes.csv as csv

# Read CSV file
db = csv.CSVFile("wt-sample.csv")

# Case number
i = 2

# Get attributes
run = db["run"][i]
pt = db["pt"][i]
mach = db["mach"][i]
alph = db["alpha"][i]
beta = db["beta"][i]

# Create a string
print("run%03i.%02i_m%.3fa%.1fb%.1f" % (run, pt, mach, alph, beta))

Included file: test02_dtypes.py

#!/usr/bin/env python
# -*- coding: utf-8 -*-

# Import CSV module
import cape.attdb.ftypes.csv as csv

# Read CSV file
db = csv.CSVFile("wt-sample.csv",
    DefaultType="float32",
    Definitions={
        "run": {"Type": "int"},
        "pt": {"Type": "int"},
    })

# Print data types
for col in db.cols:
    print("%-5s: %s" % (col, db[col].dtype.name))
